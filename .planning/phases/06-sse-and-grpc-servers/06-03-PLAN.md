<objective>
Build the SSE output server: SSEConsumer implementing router.Consumer plus SSEServer http.Handler with Last-Event-ID resume, ping keepalive, and CORS.

Purpose: SSE gives browser-native and curl-accessible streaming without gRPC tooling. Multiple consumers connect independently; each gets their own event stream rooted at their persisted cursor. Proxy-keepalive pings prevent long-lived connections from being killed by load balancers after 60-90s idle.

Output: `internal/output/sse/consumer.go` and `internal/output/sse/server.go`.

    // SSEConsumer implements router.Consumer for a single SSE connection.
    // Created per-connection in SSEServer.ServeHTTP.
    //
    // Lifecycle: Register with router -> Deliver events until client disconnects
    // -> Deliver returns io.ErrClosedPipe or write error -> RetryScheduler
    // dead-letters the consumer (isPermanentError path). No explicit Deregister
    // needed — permanent error self-deregisters via dead-letter.
    type SSEConsumer struct {
        id     string              // stable: "sse:<consumerID>"
        w      http.ResponseWriter
        rc     *http.ResponseController
        filter *output.EventFilter
        cs     router.ConsumerCursorStore
        m      *observability.KaptantoMetrics
    }

    func NewSSEConsumer(consumerID string, w http.ResponseWriter,
        filter *output.EventFilter,
        cs router.ConsumerCursorStore,
        m *observability.KaptantoMetrics) *SSEConsumer {
        return &SSEConsumer{
            id:     "sse:" + consumerID,
            w:      w,
            rc:     http.NewResponseController(w),
            filter: filter,
            cs:     cs,
            m:      m,
        }
    }

    func (c *SSEConsumer) ID() string { return c.id }

    func (c *SSEConsumer) Deliver(ctx context.Context, entry eventlog.LogEntry) error {
        // Filtered events: advance cursor silently, no write to wire.
        if !c.filter.Allow(entry.Event) {
            return c.cs.SaveCursor(ctx, c.id, entry.PartitionID, entry.Seq)
        }

        // SSE wire format: id line + data line + blank line terminator
        if _, err := fmt.Fprintf(c.w, "id: %s\ndata: ", entry.Event.ID.String()); err != nil {
            return err // broken pipe → isPermanentError → dead-letter
        }
        if err := json.NewEncoder(c.w).Encode(entry.Event); err != nil {
            return err
        }
        if _, err := fmt.Fprint(c.w, "\n"); err != nil {
            return err
        }
        if err := c.rc.Flush(); err != nil {
            return err
        }

        // Persist cursor after successful delivery.
        if err := c.cs.SaveCursor(ctx, c.id, entry.PartitionID, entry.Seq+1); err != nil {
            return err // treat cursor save failure as delivery failure
        }

        // Record metric: events delivered by this consumer.
        if c.m != nil {
            c.m.EventsDelivered.WithLabelValues(c.id, entry.Event.Table, string(entry.Event.Operation)).Inc()
        }
        return nil
    }
    ```

    IMPORTANT: `entry.PartitionID` — check if LogEntry already has a PartitionID field. If not, the partition is implicit from the goroutine context. Look at eventlog.LogEntry definition. If PartitionID is not on LogEntry, SaveCursor can use partitionID=0 for now (a follow-up can add it when needed); document with a TODO comment.

    Note: `http.NewResponseController` is the correct API (not the deprecated `w.(http.Flusher)` type assertion). It works correctly with `httptest.ResponseRecorder` in tests.

    // SSEServer is an http.Handler that manages SSE connections.
    // Each GET /stream request creates an independent SSEConsumer registered with the Router.
    type SSEServer struct {
        router       *router.Router
        cursorStore  router.ConsumerCursorStore
        metrics      *observability.KaptantoMetrics
        corsOrigin   string        // e.g. "*" or "https://example.com"
        pingInterval time.Duration // default 15s
    }

    func NewSSEServer(r *router.Router, cs router.ConsumerCursorStore,
        m *observability.KaptantoMetrics,
        corsOrigin string, pingInterval time.Duration) *SSEServer {
        if pingInterval == 0 {
            pingInterval = 15 * time.Second
        }
        if corsOrigin == "" {
            corsOrigin = "*"
        }
        return &SSEServer{
            router: r, cursorStore: cs, metrics: m,
            corsOrigin: corsOrigin, pingInterval: pingInterval,
        }
    }

    func (s *SSEServer) ServeHTTP(w http.ResponseWriter, r *http.Request) {
        // Set SSE headers BEFORE any write (first write flushes headers implicitly).
        // CRITICAL: Must happen before router.Register which triggers immediate Deliver.
        w.Header().Set("Content-Type", "text/event-stream")
        w.Header().Set("Cache-Control", "no-cache")
        w.Header().Set("Connection", "keep-alive")
        w.Header().Set("Access-Control-Allow-Origin", s.corsOrigin)

        consumerID := r.URL.Query().Get("consumer")
        if consumerID == "" {
            consumerID = r.RemoteAddr // fallback: remote addr as consumer ID
        }

        // Parse filter from query params: ?tables=orders,payments&operations=insert,update
        tables := strings.Split(r.URL.Query().Get("tables"), ",")
        ops := strings.Split(r.URL.Query().Get("operations"), ",")
        // Remove empty strings from split
        tables = filterNonEmpty(tables)
        ops = filterNonEmpty(ops)

        filter := output.NewEventFilter(tables, ops)
        consumer := NewSSEConsumer(consumerID, w, filter, s.cursorStore, s.metrics)

        // Last-Event-ID: consumerID drives cursor lookup (NOT the ULID directly).
        // The cursor store already has the correct (partitionID, seq) from the
        // previous connection's SaveCursor calls. Last-Event-ID is used only to
        // confirm the consumer wants to resume (not to address the EventLog).
        // The cursor store LoadCursor is called by the Router's runPartition loop
        // using consumer.ID() when Register is called.
        lastEventID := r.Header.Get("Last-Event-ID")
        _ = lastEventID // cursor is loaded by Router from cursor store by consumerID

        s.router.Register(consumer)

        pingTicker := time.NewTicker(s.pingInterval)
        defer pingTicker.Stop()

        rc := http.NewResponseController(w)
        for {
            select {
            case <-r.Context().Done():
                return
            case <-pingTicker.C:
                fmt.Fprint(w, ": ping\n\n")
                rc.Flush() // ignore flush error on ping — next Deliver will catch it
            }
        }
    }

    func filterNonEmpty(ss []string) []string {
        out := ss[:0]
        for _, s := range ss {
            if s != "" {
                out = append(out, s)
            }
        }
        return out
    }
    ```
