<objective>
Build the gRPC output server: proto definition, generated stubs, GRPCConsumer with channel bridge (avoids holding Router lock during stream.Send), and GRPCServer implementing Subscribe and Acknowledge RPCs.

Purpose: gRPC gives type-safe, language-agnostic streaming with HTTP/2 native backpressure. The Subscribe RPC is server-streaming; the Acknowledge RPC lets clients confirm delivery and persist their cursor. HTTP/2 flow control means stream.Send() blocks automatically when the client stops reading — no manual rate limiting needed.

Output: `internal/output/grpc/` package with proto, consumer, server, and tests.
</objective>

    ```protobuf
    syntax = "proto3";
    package kaptanto.v1;
    option go_package = "github.com/kaptanto/kaptanto/internal/output/grpc/proto";

    service CdcStream {
      // Subscribe opens a server-streaming RPC delivering ChangeEvents to the caller.
      // Events are filtered by tables and operations in SubscribeRequest.
      rpc Subscribe(SubscribeRequest) returns (stream ChangeEvent);

      // Acknowledge advances the consumer's durable cursor to the given event ID.
      // Call after successfully processing an event to persist the resume position.
      rpc Acknowledge(AcknowledgeRequest) returns (AcknowledgeResponse);
    }

    message SubscribeRequest {
      string consumer_id         = 1; // stable consumer identifier
      repeated string tables     = 2; // empty = all tables (CFG-03)
      repeated string operations = 3; // empty = all operations (CFG-04)
    }

    message ChangeEvent {
      string id              = 1; // ULID string
      string idempotency_key = 2;
      string operation       = 3; // insert | update | delete | read | control
      string table           = 4;
      bytes  payload         = 5; // full JSON-encoded ChangeEvent for simplicity (OUT-07)
    }

    message AcknowledgeRequest {
      string consumer_id   = 1;
      string event_id      = 2; // ULID of the last processed event
      uint32 partition_id  = 3; // partition to advance cursor on
      uint64 seq           = 4; // Event Log sequence to advance to
    }

    message AcknowledgeResponse {
      bool ok = 1;
    }
    ```

    `internal/output/grpc/consumer.go`:
    ```go
    package grpcoutput

    // GRPCConsumer implements router.Consumer via a channel bridge.
    // Deliver() sends to a buffered channel (non-blocking); the Subscribe
    // goroutine reads from the channel and calls stream.Send() OUTSIDE any
    // Router lock — preventing deadlock from HTTP/2 backpressure (OUT-08).
    type GRPCConsumer struct {
        id     string // stable: "grpc:<consumerID>"
        ch     chan *proto.ChangeEvent
        filter *output.EventFilter
        cs     router.ConsumerCursorStore
        m      *observability.KaptantoMetrics
        done   chan struct{} // closed when Subscribe handler exits
    }

    func NewGRPCConsumer(consumerID string, bufSize int,
        filter *output.EventFilter,
        cs router.ConsumerCursorStore,
        m *observability.KaptantoMetrics) *GRPCConsumer {
        return &GRPCConsumer{
            id:     "grpc:" + consumerID,
            ch:     make(chan *proto.ChangeEvent, bufSize), // bufSize=64 default
            filter: filter,
            cs:     cs,
            m:      m,
            done:   make(chan struct{}),
        }
    }

    func (c *GRPCConsumer) ID() string { return c.id }

    func (c *GRPCConsumer) Deliver(ctx context.Context, entry eventlog.LogEntry) error {
        if !c.filter.Allow(entry.Event) {
            return nil // filtered events: no delivery, cursor advances via Router
        }

        // Encode to proto ChangeEvent: payload carries full JSON for OUT-07
        payload, err := json.Marshal(entry.Event)
        if err != nil {
            return fmt.Errorf("grpc consumer: marshal event: %w", err)
        }
        protoEv := &proto.ChangeEvent{
            Id:             entry.Event.ID.String(),
            IdempotencyKey: entry.Event.IdempotencyKey,
            Operation:      string(entry.Event.Operation),
            Table:          entry.Event.Table,
            Payload:        payload,
        }

        // Non-blocking send: if channel is full, return error so RetryScheduler backs off.
        select {
        case c.ch <- protoEv:
            return nil
        case <-c.done:
            return fmt.Errorf("grpc consumer: subscribe handler exited")
        default:
            return fmt.Errorf("grpc consumer: channel full, slow client")
        }
    }

    // Close signals that the Subscribe handler has exited. Called by Subscribe on return.
    func (c *GRPCConsumer) Close() { close(c.done) }
    ```

    `internal/output/grpc/server.go`:
    ```go
    package grpcoutput

    import (
        grpclib "google.golang.org/grpc"
        "google.golang.org/grpc/keepalive"
        "google.golang.org/grpc/codes"
        "google.golang.org/grpc/status"
    )

    // GRPCServer implements proto.CdcStreamServer and manages gRPC connections.
    type GRPCServer struct {
        proto.UnimplementedCdcStreamServer
        router      *router.Router
        cursorStore router.ConsumerCursorStore
        metrics     *observability.KaptantoMetrics
    }

    func NewGRPCServer(r *router.Router, cs router.ConsumerCursorStore,
        m *observability.KaptantoMetrics) *GRPCServer {
        return &GRPCServer{router: r, cursorStore: cs, metrics: m}
    }

    // NewGRPCNetServer creates and configures the grpc.Server. Call Serve(lis) on it.
    func NewGRPCNetServer(svc *GRPCServer) *grpclib.Server {
        srv := grpclib.NewServer(
            grpclib.MaxConcurrentStreams(1000),
            grpclib.KeepaliveParams(keepalive.ServerParameters{
                Time:    30 * time.Second,
                Timeout: 10 * time.Second,
            }),
        )
        proto.RegisterCdcStreamServer(srv, svc)
        return srv
    }

    func (s *GRPCServer) Subscribe(req *proto.SubscribeRequest,
        stream proto.CdcStream_SubscribeServer) error {

        filter := output.NewEventFilter(req.Tables, req.Operations)
        consumer := NewGRPCConsumer(req.ConsumerId, 64, filter, s.cursorStore, s.metrics)
        defer consumer.Close() // signals Deliver that handler has exited

        s.router.Register(consumer)

        ctx := stream.Context()
        for {
            select {
            case <-ctx.Done():
                return status.FromContextError(ctx.Err()).Err()
            case ev, ok := <-consumer.ch:
                if !ok {
                    return nil
                }
                // stream.Send blocks when HTTP/2 window is exhausted (OUT-08).
                // Called here OUTSIDE any Router or RetryScheduler lock.
                if err := stream.Send(ev); err != nil {
                    return err
                }
            }
        }
    }

    func (s *GRPCServer) Acknowledge(ctx context.Context,
        req *proto.AcknowledgeRequest) (*proto.AcknowledgeResponse, error) {

        consumerID := "grpc:" + req.ConsumerId
        if err := s.cursorStore.SaveCursor(ctx, consumerID, req.PartitionId, req.Seq); err != nil {
            return nil, status.Errorf(codes.Internal, "save cursor: %v", err)
        }
        return &proto.AcknowledgeResponse{Ok: true}, nil
    }
    ```
