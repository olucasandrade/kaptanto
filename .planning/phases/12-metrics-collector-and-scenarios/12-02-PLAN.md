VmRSS host-proc approach (avoids distroless exec failure, see Pitfall 5):
```go
func readVmRSSFromHost(containerName string) (int64, error) {
    pidOut, err := exec.Command("docker", "inspect",
        "--format", "{{.State.Pid}}", containerName).Output()
    if err != nil {
        return 0, err
    }
    pid := strings.TrimSpace(string(pidOut))
    data, err := os.ReadFile(fmt.Sprintf("/proc/%s/status", pid))
    if err != nil {
        return 0, err
    }
    for _, line := range strings.Split(string(data), "\n") {
        if strings.HasPrefix(line, "VmRSS:") {
            fields := strings.Fields(line)
            if len(fields) >= 2 {
                return strconv.ParseInt(fields[1], 10, 64)
            }
        }
    }
    return 0, fmt.Errorf("VmRSS not found in /proc/%s/status", pid)
}
```
